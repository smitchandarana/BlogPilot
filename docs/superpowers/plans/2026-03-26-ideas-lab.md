# Ideas Lab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an "Ideas Lab" tab to Content Studio where the user can browse posts from all sources (LinkedIn, Reddit, RSS, own posts), pin up to 5, tag excerpts by type, auto-generate an editable synthesis brief, and generate a final post — inline or by handing off to the existing Generator tab.

**Architecture:** Three new backend endpoints added to `backend/api/content.py` (idea-pool search, synthesize-brief via Groq, generate-from-brief). Three new frontend components (`IdeasLab.jsx`, `IdeaPoolPanel.jsx`, `MixBoard.jsx`) wired into `ContentStudio.jsx` via a new top-level tab bar. All state lives in `IdeasLab.jsx`; pinned items and brief survive tab switches within the session.

**Tech Stack:** Python/FastAPI (backend), React 18 + Tailwind (frontend), Groq via existing `build_ai_client("generation")`, existing `PromptLoader` pattern.

**Spec:** `docs/superpowers/specs/2026-03-26-ideas-lab-design.md`

```python
@router.get("/idea-pool")
async def get_idea_pool(
    q: Optional[str] = Query(None),
    source: str = Query("all"),
    topic: Optional[str] = Query(None),
    limit: int = Query(30),
    offset: int = Query(0),
):
    """
    Return a unified, normalized list of content items for the Ideas Lab.
    Sources: linkedin (posts table), reddit/rss/hn (research_snippets), my_posts (scheduled_posts).
    Ranked by keyword overlap with `topic` if provided, otherwise by recency.
    """
    from backend.storage.models import Post, ResearchSnippet, ScheduledPost

    items = []

    with get_db() as db:
        # ── LinkedIn posts ──────────────────────────────────────────────
        if source in ("all", "linkedin"):
            rows = db.query(Post).filter(Post.text.isnot(None)).order_by(
                Post.created_at.desc()
            ).limit(200).all()
            for r in rows:
                items.append({
                    "id": f"li_{r.id}",
                    "source_type": "LINKEDIN",
                    "title": r.author_name,
                    "text": r.text or "",
                    "url": r.url,
                    "date": r.created_at.isoformat() if r.created_at else None,
                    "engagement_score": float((r.like_count or 0) + (r.comment_count or 0) * 2),
                })

        # ── Research snippets (Reddit / RSS / HN) ───────────────────────
        if source in ("all", "reddit", "rss", "hn"):
            source_filter = None
            if source == "reddit":
                source_filter = "REDDIT"
            elif source == "rss":
                source_filter = "RSS"
            elif source == "hn":
                source_filter = "HN"

            q_snippets = db.query(ResearchSnippet).filter(
                ResearchSnippet.snippet.isnot(None)
            )
            if source_filter:
                q_snippets = q_snippets.filter(
                    ResearchSnippet.source == source_filter
                )
            rows = q_snippets.order_by(
                ResearchSnippet.engagement_signal.desc()
            ).limit(200).all()
            for r in rows:
                items.append({
                    "id": f"rs_{r.id}",
                    "source_type": r.source or "RSS",
                    "title": r.title,
                    "text": r.snippet or "",
                    "url": r.source_url,
                    "date": r.discovered_at.isoformat() if r.discovered_at else None,
                    "engagement_score": float(r.engagement_signal or 0),
                })

        # ── My generated/published posts ────────────────────────────────
        if source in ("all", "my_posts"):
            rows = db.query(ScheduledPost).filter(
                ScheduledPost.status.in_(["PUBLISHED", "SCHEDULED"])
            ).order_by(ScheduledPost.created_at.desc()).limit(100).all()
            for r in rows:
                items.append({
                    "id": f"sp_{r.id}",
                    "source_type": "MY_POST",
                    "title": r.topic,
                    "text": r.text or "",
                    "url": None,
                    "date": r.created_at.isoformat() if r.created_at else None,
                    "engagement_score": 0.0,
                })

    # ── Keyword search ──────────────────────────────────────────────────
    if q:
        q_lower = q.lower()
        items = [
            i for i in items
            if q_lower in (i["text"] or "").lower()
            or q_lower in (i["title"] or "").lower()
        ]

    # ── Topic relevance ranking ─────────────────────────────────────────
    if topic:
        topic_tokens = set(topic.lower().split())

        def _score(item):
            combined = f"{item.get('title', '') or ''} {item.get('text', '') or ''}".lower()
            return sum(1 for t in topic_tokens if t in combined)

        items.sort(key=_score, reverse=True)
    else:
        # Default: recency (already ordered per-source, stable sort preserves it)
        pass

    # Pagination
    total = len(items)
    page = items[offset: offset + limit]
    return page
```

```python
@router.post("/synthesize-brief")
async def synthesize_brief(body: SynthesizeBriefRequest):
    """
    Call Groq to assemble a synthesis brief from pinned posts + tagged highlights.
    If highlights exist for a selection, they are used. Otherwise falls back to
    first 300 chars of full_text.
    """
    if not body.selections:
        raise HTTPException(status_code=422, detail="At least one selection is required")

    try:
        from backend.ai.prompt_loader import PromptLoader
        groq_client = build_ai_client("generation")
        prompt_loader = PromptLoader()
        prompt_loader.load_all()
    except AIClientUnavailableError as e:
        raise HTTPException(status_code=503, detail=str(e))
    except Exception as e:
        raise HTTPException(status_code=503, detail=f"AI setup failed: {e}")

    # Build materials block
    materials_parts = []
    for i, sel in enumerate(body.selections, 1):
        source_label = sel.source_type.replace("_", " ").title()
        if sel.highlights:
            tagged_lines = "\n".join(
                f"  [{h.tag.upper()}] {h.text}" for h in sel.highlights
            )
            materials_parts.append(
                f"Source {i} ({source_label}) — tagged excerpts:\n{tagged_lines}"
            )
        else:
            fallback = (sel.full_text or "")[:300]
            materials_parts.append(
                f"Source {i} ({source_label}) — full text excerpt:\n  {fallback}"
            )

    materials_block = "\n\n".join(materials_parts)

    try:
        prompt = prompt_loader.format(
            "synthesize_brief",
            source_count=len(body.selections),
            materials=materials_block,
        )
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Prompt formatting failed: {e}")

    _system = (
        "You are a LinkedIn content strategist. "
        "Return ONLY the synthesis brief. No preamble. No markdown."
    )

    try:
        brief = await groq_client.complete(_system, prompt)
        return {"brief": brief.strip()}
    except Exception as e:
        logger.error(f"synthesize-brief: Groq call failed — {e}")
        raise HTTPException(status_code=503, detail=f"AI call failed: {e}")
```

```python
@router.post("/generate-from-brief")
async def generate_from_brief(body: GenerateFromBriefRequest):
    """
    Generate a LinkedIn post from a synthesis brief.
    Injects the brief as core_insight into the existing structured_post prompt.
    """
    if not body.brief or not body.brief.strip():
        raise HTTPException(status_code=422, detail="Brief cannot be empty")

    try:
        from backend.ai.prompt_loader import PromptLoader
        groq_client = build_ai_client("generation")
        prompt_loader = PromptLoader()
        prompt_loader.load_all()
    except AIClientUnavailableError as e:
        raise HTTPException(status_code=503, detail=str(e))
    except Exception as e:
        raise HTTPException(status_code=503, detail=f"AI setup failed: {e}")

    from backend.ai import post_generator
    structured_inputs = {
        "topic": body.topic,
        "subtopic": body.topic,
        "pain_point": "",
        "audience": "",
        "hook_intent": "STORY",
        "core_insight": body.brief,   # brief becomes the core_insight
        "belief_to_challenge": "",
        "proof_type": "EXAMPLE",
        "real_scenario": "",
        "style": body.style,
        "tone": body.tone,
        "word_count": body.word_count,
    }

    result = await post_generator.generate_structured(
        structured_inputs=structured_inputs,
        groq_client=groq_client,
        prompt_loader=prompt_loader,
        evidence="",
    )
    return result
```

```jsx
 * IdeaPoolPanel — left panel of Ideas Lab.

  const [loading, setLoading] = useState(false)
```

```jsx
import { X, Loader2, Copy, Check, ExternalLink, Wand2, Send } from 'lucide-react'
import { content as contentApi } from '../api/client'

const TAG_STYLES = {
  Hook: 'bg-violet-500/20 text-violet-300 border-violet-500/30',
  Stat: 'bg-sky-500/20 text-sky-300 border-sky-500/30',
  Story: 'bg-amber-500/20 text-amber-300 border-amber-500/30',
  Insight: 'bg-emerald-500/20 text-emerald-300 border-emerald-500/30',
  Example: 'bg-rose-500/20 text-rose-300 border-rose-500/30',
}

const TAG_UNDERLINE = {
  Hook: 'underline decoration-violet-400 decoration-2',
  Stat: 'underline decoration-sky-400 decoration-2',
  Story: 'underline decoration-amber-400 decoration-2',
  Insight: 'underline decoration-emerald-400 decoration-2',
  Example: 'underline decoration-rose-400 decoration-2',
}

const STYLES = ['Thought Leadership', 'Story', 'Tips List', 'Question', 'Data Insight', 'Contrarian Take']
const TONES = ['Professional', 'Conversational', 'Bold', 'Educational']
const TAGS = ['Hook', 'Stat', 'Story', 'Insight', 'Example']

/**
 * MixBoard — right panel of Ideas Lab.
 * Props:
 *   pinnedItems: array of {id, source_type, title, text, highlights: [{text, tag}]}
 *   onUnpin: (id) => void
 *   onUpdateHighlights: (id, highlights) => void
 *   onSendToGenerator: (brief) => void
 */
export default function MixBoard({ pinnedItems, onUnpin, onUpdateHighlights, onSendToGenerator }) {
  const [brief, setBrief] = useState('')
  const [briefLoading, setBriefLoading] = useState(false)
  const [briefError, setBriefError] = useState(null)
  const [style, setStyle] = useState('Thought Leadership')
  const [tone, setTone] = useState('Professional')
  const [wordCount, setWordCount] = useState(150)
  const [generating, setGenerating] = useState(false)
  const [inlineResult, setInlineResult] = useState(null)
  const [generateError, setGenerateError] = useState(null)
  const [copied, setCopied] = useState(false)

  // Text selection tagging state
  const [tagPopover, setTagPopover] = useState(null) // {x, y, selectedText, itemId}
  const boardRef = useRef(null)

  // ── Brief building ───────────────────────────────────────────────────────
  const handleBuildBrief = async () => {
    setBriefLoading(true)
    setBriefError(null)
    try {
      const selections = pinnedItems.map((item) => ({
        id: item.id,
        source_type: item.source_type,
        full_text: item.text,
        highlights: item.highlights || [],
      }))
      const res = await contentApi.synthesizeBrief(selections)
      setBrief(res.data.brief || '')
    } catch (err) {
      setBriefError(err.response?.data?.detail || 'Failed to build brief — try again')
    } finally {
      setBriefLoading(false)
    }
  }

  // ── Generate ─────────────────────────────────────────────────────────────
  const handleGenerate = async () => {
    setGenerating(true)
    setGenerateError(null)
    setInlineResult(null)
    try {
      const res = await contentApi.generateFromBrief({
        brief,
        topic: pinnedItems[0]?.title || 'Business Intelligence',
        style,
        tone,
        word_count: wordCount,
      })
      setInlineResult(res.data?.post || res.data?.text || '')
    } catch (err) {
      setGenerateError(err.response?.data?.detail || 'Generation failed — try again')
    } finally {
      setGenerating(false)
    }
  }

  // ── Copy ─────────────────────────────────────────────────────────────────
  const handleCopy = useCallback(() => {
    if (!inlineResult) return
    navigator.clipboard.writeText(inlineResult).then(() => {
      setCopied(true)
      setTimeout(() => setCopied(false), 2000)
    })
  }, [inlineResult])

  // ── Text selection → tag popover ─────────────────────────────────────────
  const handleMouseUp = (itemId) => (e) => {
    const selection = window.getSelection()
    const selectedText = selection?.toString().trim()
    if (!selectedText || selectedText.length < 5) {
      setTagPopover(null)
      return
    }
    const range = selection.getRangeAt(0)
    const rect = range.getBoundingClientRect()
    const boardRect = boardRef.current?.getBoundingClientRect() || { left: 0, top: 0 }
    setTagPopover({
      x: rect.left - boardRect.left + rect.width / 2,
      y: rect.top - boardRect.top - 8,
      selectedText,
      itemId,
    })
  }

  const handleApplyTag = (tag) => {
    if (!tagPopover) return
    const item = pinnedItems.find((i) => i.id === tagPopover.itemId)
    if (!item) return
    const existing = item.highlights || []
    // Avoid double-tagging exact same text
    if (existing.some((h) => h.text === tagPopover.selectedText)) {
      setTagPopover(null)
      return
    }
    onUpdateHighlights(tagPopover.itemId, [
      ...existing,
      { text: tagPopover.selectedText, tag },
    ])
    setTagPopover(null)
    window.getSelection()?.removeAllRanges()
  }

  const handleRemoveHighlight = (itemId, highlightText) => {
    const item = pinnedItems.find((i) => i.id === itemId)
    if (!item) return
    onUpdateHighlights(itemId, (item.highlights || []).filter((h) => h.text !== highlightText))
  }

  // ── Render card text with highlights ─────────────────────────────────────
  const renderHighlightedText = (item) => {
    const highlights = item.highlights || []
    if (!highlights.length) return <span>{item.text}</span>

    let result = item.text
    // Build spans by scanning for highlight text
    const parts = []
    let remaining = result
    highlights.forEach(({ text: ht, tag }) => {
      const idx = remaining.indexOf(ht)
      if (idx === -1) return
      if (idx > 0) parts.push({ text: remaining.slice(0, idx), tag: null })
      parts.push({ text: ht, tag })
      remaining = remaining.slice(idx + ht.length)
    })
    if (remaining) parts.push({ text: remaining, tag: null })

    return (
      <>
        {parts.map((p, i) =>
          p.tag ? (
            <span
              key={i}
              className={`cursor-pointer ${TAG_UNDERLINE[p.tag] || ''}`}
              title={`${p.tag} — click to remove`}
              onClick={() => handleRemoveHighlight(item.id, p.text)}
            >
              {p.text}
            </span>
          ) : (
            <span key={i}>{p.text}</span>
          )
        )}
      </>
    )
  }

  const canBuildBrief = pinnedItems.length > 0
  const canGenerate = brief.trim().length > 0

  return (
    <div ref={boardRef} className="relative flex h-full flex-col gap-4">
      {/* ── Tag popover ──────────────────────────────────────────────── */}
      {tagPopover && (
        <div
          className="absolute z-50 flex gap-1 rounded-lg border border-slate-600 bg-slate-800 p-1.5 shadow-xl"
          style={{ left: tagPopover.x - 80, top: tagPopover.y - 44 }}
        >
          {TAGS.map((tag) => (
            <button
              key={tag}
              onClick={() => handleApplyTag(tag)}
              className={`rounded px-2 py-0.5 text-[11px] font-medium border ${TAG_STYLES[tag]}`}
            >
              {tag}
            </button>
          ))}
          <button
            onClick={() => setTagPopover(null)}
            className="ml-1 text-slate-500 hover:text-slate-300"
          >
            <X className="h-3 w-3" />
          </button>
        </div>
      )}

      {/* ── Zone 1: Pinned posts ──────────────────────────────────────── */}
      <div className="flex-none">
        <h3 className="mb-2 text-[10px] font-semibold uppercase tracking-widest text-slate-600">
          Pinned Posts {pinnedItems.length > 0 && `(${pinnedItems.length}/5)`}
        </h3>
        {pinnedItems.length === 0 && (
          <p className="rounded-lg border border-dashed border-slate-700/60 py-6 text-center text-xs text-slate-600">
            Pin posts from the left panel to start mixing
          </p>
        )}
        <div className="space-y-2 max-h-72 overflow-y-auto pr-1">
          {pinnedItems.map((item) => (
            <div
              key={item.id}
              className="rounded-lg border border-slate-700/40 bg-slate-800/40 p-3"
            >
              <div className="flex items-center justify-between mb-1.5">
                <span className="text-[11px] font-medium text-slate-400">
                  {item.title || item.source_type}
                </span>
                <button
                  onClick={() => onUnpin(item.id)}
                  className="text-slate-600 hover:text-slate-400 transition-colors"
                >
                  <X className="h-3.5 w-3.5" />
                </button>
              </div>
              <p
                className="text-xs text-slate-300 leading-relaxed select-text cursor-text"
                onMouseUp={handleMouseUp(item.id)}
              >
                {renderHighlightedText(item)}
              </p>
              {(item.highlights || []).length > 0 && (
                <div className="mt-2 flex flex-wrap gap-1">
                  {item.highlights.map((h, i) => (

              )}
            </div>
          ))}
        </div>
      </div>

      {/* ── Zone 2: Synthesis brief ───────────────────────────────────── */}
      <div className="flex-none">
        <div className="flex items-center justify-between mb-2">
          <h3 className="text-[10px] font-semibold uppercase tracking-widest text-slate-600">
            Synthesis Brief

            onClick={handleBuildBrief}

            className={`flex items-center gap-1 rounded px-2.5 py-1 text-xs transition-colors ${
              canBuildBrief && !briefLoading
                ? 'bg-slate-700/60 text-slate-200 hover:bg-slate-700'
                : 'text-slate-600 cursor-not-allowed'

            title={!canBuildBrief ? 'Pin at least one post to build a brief' : ''}
```

```jsx
import { useState, useCallback } from 'react'
import IdeaPoolPanel from '../components/IdeaPoolPanel'

/**
 * IdeasLab — the "Ideas Lab" tab content in Content Studio.
 * Props:
 *   onSendToGenerator: (brief) => void  — switches to Generate tab with brief pre-filled
 */
export default function IdeasLab({ onSendToGenerator }) {
  const [pinnedItems, setPinnedItems] = useState([])

  const pinnedIds = new Set(pinnedItems.map((i) => i.id))
  const canPin = pinnedItems.length < 5

  const handlePin = useCallback((item) => {
    setPinnedItems((prev) => {
      if (prev.find((p) => p.id === item.id)) return prev
      if (prev.length >= 5) return prev
      return [...prev, { ...item, highlights: [] }]
    })
  }, [])

  const handleUnpin = useCallback((id) => {
    setPinnedItems((prev) => prev.filter((i) => i.id !== id))
  }, [])

  const handleUpdateHighlights = useCallback((id, highlights) => {
    setPinnedItems((prev) =>
      prev.map((i) => (i.id === id ? { ...i, highlights } : i))
    )
  }, [])

  return (
```
